String? getIdentifierFromData<DataT extends Data>(DataT? data) {
  switch (typeOf<DataT>()) {
    case CheckCategory:
      return 'category';

    case CheckPoint:
      return 'checkpoint';

    case CheckPointDefect:
      return 'defect';

    case InspectionLocation:
      return 'location';

    default:
      debugPrint("yo this type is not supported : ${typeOf<DataT>()}");
      return null;
  }
}
